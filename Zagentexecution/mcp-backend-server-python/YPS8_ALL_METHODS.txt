  METHOD SET_SELECTION_VALUES.

    FIELD-SYMBOLS <lt_range> TYPE ANY TABLE.
    FIELD-SYMBOLS <lv_param> TYPE any.
    DATA lv_selname TYPE fieldname.

    lv_selname = iv_selname.
    CASE iv_kind.
      WHEN 'S'.   "SELECT-OPTIONS
        REPLACE 'S_' IN lv_selname WITH 'MR_'.
        ASSIGN (lv_selname) TO <lt_range>.
        CHECK <lt_range> IS ASSIGNED.
        <lt_range> = it_value.
      WHEN 'P'. "PARAMETERS
        REPLACE 'P_' IN lv_selname WITH 'MP_'.
        ASSIGN (lv_selname) TO <lv_param>.
        CHECK <lv_param> IS ASSIGNED.
        <lv_param> = iv_value.
    ENDCASE.

  ENDMETHOD.

--- Method Implementation: YCL_YPS8_BCS_BL===============CM005 ---
  METHOD prepare_data.

    DATA lv_end_fistl TYPE boolean.
    DATA ls_result TYPE ysfm_data_rep_2.
    DATA ls_result_head TYPE ysfm_data_rep_2.
    DATA lt_wfmbdt TYPE tty_wfmbdt.
    DATA lt_wfmbdt_oth TYPE tty_wfmbdt.
    DATA lt_wfmbdt_fund TYPE tty_wfmbdt.
    DATA lr_fistl TYPE tty_fistl.
    DATA lv_rate_today TYPE ukurs_curr.

    LOOP AT mt_fmbdt INTO DATA(ls_fmbdt).

      AT NEW fikrs.
        CLEAR: mv_waers, mv_convert, lv_rate_today.
        READ TABLE mo_conv_rate->mt_conv INTO DATA(ls_conv) WITH KEY fikrs = ls_fmbdt-fikrs.
        IF sy-subrc = 0.
          mv_waers = ls_conv-waers.
          mv_convert = abap_true.
          "Get rate at sy-datum
*          READ TABLE mt_conv_data INTO DATA(ls_conv_data) WITH KEY waers = mv_waers
*                                                                   gjahr = '0000'
*                                                                   perio = '000'.
          lv_rate_today = mo_conv_rate->get_rate( iv_waers = mv_waers
                                                  iv_gjahr = '0000'
                                                  iv_perio = '000' ).
        ENDIF.
      ENDAT.

      AT NEW fincode.
        CLEAR: ms_fipex_all, ls_result_head, lt_wfmbdt_fund, lr_fistl, mv_no_project.
        ls_result_head-fikrs = ls_fmbdt-fikrs.
        ls_result_head-fincode = ls_fmbdt-fincode.
        me->set_master_data( CHANGING cs_result = ls_result_head ).
      ENDAT.

      AT NEW fistl.
        CLEAR: lt_wfmbdt, lt_wfmbdt_oth, ls_result.
        mv_excl80 = mp_excl80.
        ls_result = ls_result_head.
        ls_result-fistl = ls_fmbdt-fistl.
      ENDAT.

      "Check project has been selected
      CHECK mv_no_project = abap_false.

      "Check fund / fund center / commitment item
      IF me->is_commitment_item_in_avc( iv_fikrs = ls_fmbdt-fikrs
                                        iv_fincode = ls_fmbdt-fincode
                                        iv_fistl = ls_fmbdt-fistl
                                        iv_fipex = ls_fmbdt-fipex ) = abap_true.
        me->set_main_to_light_table( EXPORTING is_fmbdt = ls_fmbdt
                                               iv_rate = lv_rate_today
                                     CHANGING ct_wfmbdt = lt_wfmbdt ).
      ENDIF.

      AT END OF fistl.
        DELETE lt_wfmbdt WHERE hamount IS INITIAL AND fipex <> 'REVENUE'.
        IF lt_wfmbdt IS NOT INITIAL.
          "Get amounts out of selection period for ALLOC_TOTAL
          LOOP AT mt_fmbdt_oth INTO DATA(ls_fmbdt_oth) WHERE fikrs = ls_fmbdt-fikrs
                                                       AND   fincode = ls_fmbdt-fincode
                                                       AND   fistl = ls_fmbdt-fistl.
            IF me->is_commitment_item_in_avc( iv_fikrs = ls_fmbdt_oth-fikrs
                                              iv_fincode = ls_fmbdt_oth-fincode
                                              iv_fistl = ls_fmbdt_oth-fistl
                                              iv_fipex = ls_fmbdt_oth-fipex ) = abap_true.
              me->set_main_to_light_table( EXPORTING is_fmbdt = ls_fmbdt_oth
                                                     iv_rate = lv_rate_today
                                           CHANGING ct_wfmbdt = lt_wfmbdt_oth ).
            ENDIF.
          ENDLOOP.
          DELETE lt_wfmbdt_oth WHERE hamount IS INITIAL AND fipex <> 'REVENUE'.
          "Set amounts in result table
          me->set_amounts( is_result = ls_result
                           it_wfmbdt = lt_wfmbdt
                           it_wfmbdt_oth = lt_wfmbdt_oth ).
          APPEND VALUE #( sign = 'I' option = 'EQ' low = ls_fmbdt-fistl ) TO lr_fistl.
          APPEND LINES OF lt_wfmbdt TO lt_wfmbdt_fund.
        ENDIF.
      ENDAT.

      AT END OF fincode.
        ls_result = ls_result_head.
        mv_excl80 = abap_false.
        me->set_other_fund_center( is_result = ls_result
                                   ir_fistl = lr_fistl
                                   it_wfmbdt = lt_wfmbdt_fund ).
        me->set_project_amounts( iv_fikrs = ls_fmbdt-fikrs
                                 iv_fincode = ls_fmbdt-fincode ).
      ENDAT.

    ENDLOOP.

  ENDMETHOD.

  METHOD read_fmbdt.

    DATA lr_fipex TYPE RANGE OF fm_fipex.
    DATA lt_fmbdt TYPE tty_fmbdt.

    lr_fipex = VALUE #( ( sign = 'E' option = 'EQ' low = 'REVENUE' )
                        ( sign = 'E' option = 'EQ' low = 'RESERVE' ) ).
    APPEND LINES OF mr_fipex TO lr_fipex.

    "Get data from FMBDT
    SELECT a~rfikrs AS fikrs,
           a~rfund AS fincode,
           a~rfundsctr AS fistl,
           a~rcmmtitem AS fipex,
           c~curr1 AS waers,
           a~budtype_9 AS btype,
           b~bukrs AS bukrs,
           a~hslvt AS hslvt,
           a~hsl01 AS hsl01,
           a~hsl02 AS hsl02,
           a~hsl03 AS hsl03,
           a~hsl04 AS hsl04,
           a~hsl05 AS hsl05,
           a~hsl06 AS hsl06,
           a~hsl07 AS hsl07,
           a~hsl08 AS hsl08,
           a~hsl09 AS hsl09,
           a~hsl10 AS hsl10,
           a~hsl11 AS hsl11,
           a~hsl12 AS hsl12,
           a~hsl13 AS hsl13,
           a~hsl14 AS hsl14,
           a~hsl15 AS hsl15,
           a~hsl16 AS hsl16,
           a~rldnr AS rldnr,
           a~rrcty AS rrcty,
           a~rvers AS rvers,
           a~ryear AS ryear,
           a~robjnr AS robjnr,
           a~cobjnr AS cobjnr,
           a~sobjnr AS sobjnr,
           a~rtcur  AS rtcur
           FROM fmbdt AS a
           LEFT OUTER JOIN t001 AS b ON b~fikrs = a~rfikrs
           LEFT OUTER JOIN t882 AS c ON  c~bukrs = b~bukrs
                                     AND c~rldnr = '00'
           FOR ALL ENTRIES IN @mt_fund
           WHERE a~rfikrs = @mt_fund-fikrs
           AND   a~rfund = @mt_fund-fincode
           AND   a~rvers = @mp_versi
           "AND   a~ryear IN @mr_gjahr
           AND   a~rfundsctr IN @mr_fictr
           AND   a~rcmmtitem IN @lr_fipex
           AND   ( a~valtype_9 = 'B1' OR a~valtype_9 = 'R1' )
           AND   a~wfstate_9 = 'P'
           AND   ( a~budtype_9 = '3000' OR a~budtype_9 = '4000' )
           INTO TABLE @lt_fmbdt.

    SORT lt_fmbdt.

    LOOP AT lt_fmbdt INTO DATA(ls_fmbdt).
      IF ls_fmbdt-ryear IN mr_gjahr.
        APPEND ls_fmbdt TO mt_fmbdt.
      ELSE.
        APPEND ls_fmbdt TO mt_fmbdt_oth.
      ENDIF.
    ENDLOOP.

  ENDMETHOD.


--- Method Implementation: YCL_YPS8_BCS_BL===============CM00A ---
  METHOD read_fmfincode.

    DATA lt_ibf TYPE RANGE OF ye_fm_c5_contribution.

    "Set IBF flag in range
    IF mp_ibf = abap_true.
      APPEND VALUE #( sign = 'I' option = 'EQ' low = abap_true ) TO lt_ibf.
    ENDIF.

    "Get funds from select criteria
    SELECT DISTINCT a~fikrs,
                    a~fincode,
                    b~bukrs,
                    a~datab,
                    a~datbis,
                    c~pspnr
           FROM fmfincode AS a
           LEFT OUTER JOIN t001 AS b ON b~fikrs = a~fikrs
           LEFT OUTER JOIN proj AS c ON  c~pspid = a~fincode
                                     AND c~vbukr = b~bukrs
           WHERE a~fikrs IN @mr_fikrs
           AND   a~fincode IN @mr_fund
           AND   a~datab <= @mv_endda
           AND   a~datbis >= @mv_begda
           AND   a~type IN @mr_type
           INTO CORRESPONDING FIELDS OF TABLE @mt_fund.

    "Keep only funds which exist as project
    DELETE mt_fund WHERE pspnr IS INITIAL.

    CHECK mt_fund IS NOT INITIAL.

    "Get fund assignment to C/5 period and Part of IBF and Output
    SELECT a~fikrs, a~fincode, a~c5_id, a~c5_sel, a~fm_output, b~year_from, b~year_to
           FROM ytfm_fund_c5 AS a INNER JOIN ytfm_c5 AS b ON b~c5_id = a~c5_id
           FOR ALL ENTRIES IN @mt_fund
           WHERE a~fikrs = @mt_fund-fikrs
           AND   a~fincode = @mt_fund-fincode
           AND   a~fm_output IN @mr_output
           AND   a~c5_id IN @mr_c5_id
           AND   a~c5_sel IN @lt_ibf
           AND   b~year_from <= @mv_endye
           AND   b~year_to >= @mv_begye
           INTO TABLE @mt_fund_c5.

    "Check selection critera related to fund assignment to C/5
    IF mr_output IS NOT INITIAL OR mr_c5_id IS NOT INITIAL OR lt_ibf IS NOT INITIAL.
      LOOP AT mt_fund INTO DATA(ls_fund).
        READ TABLE mt_fund_c5 TRANSPORTING NO FIELDS WITH KEY fikrs = ls_fund-fikrs
                                                              fincode = ls_fund-fincode.
        CHECK sy-subrc <> 0.
        DELETE mt_fund.
      ENDLOOP.
    ENDIF.

    "Get output masterdata
    SELECT a~fm_output,
           a~otype,
           b~otext
           FROM ytfm_output AS a
           LEFT OUTER JOIN ytfm_output_t AS b ON  b~sprsl = @sy-langu
                                              AND b~fm_output = a~fm_output
           WHERE a~fm_output IN @mr_output
           INTO TABLE @mt_output.

  ENDMETHOD.

  METHOD filter_fund.

    DATA lt_fund TYPE table of ty_fund.

    LOOP AT mt_fmbdt INTO DATA(ls_fmbdt).
      AT NEW fincode.

  METHOD set_amounts.

    DATA ls_wfmbdt TYPE ty_wfmbdt.
    DATA ls_wfmbdt_oth TYPE ty_wfmbdt.
    FIELD-SYMBOLS <ls_wfmbdt> TYPE ty_wfmbdt.
    FIELD-SYMBOLS <ls_wfmbdt_oth> TYPE ty_wfmbdt.
    DATA lv_fipex_80 TYPE boolean.
    DATA lr_fipex TYPE tty_fipex.
    DATA ls_fmit TYPE ty_fmit.
    DATA ls_result_initial TYPE ysfm_data_rep_2.
    DATA ls_result_usd TYPE ysfm_data_rep_2.
    DATA lv_whgbtr_usd TYPE whgxxx.
    DATA lv_fipex_high TYPE fm_fipex VALUE '80'.

    "save initial values
    ls_result_initial = is_result.
    ls_result_usd = is_result.

    """"""""""""""""""""""""""""""""""""""""""""""""""
    """""""""""Set data dependant of commitment item
    """"""""""""""""""""""""""""""""""""""""""""""""""
    "Check if exists Commitment item with value 80
    READ TABLE it_wfmbdt TRANSPORTING NO FIELDS WITH KEY fipex = '80'.
    IF sy-subrc = 0.
      lv_fipex_80 = abap_true.
      IF mv_excl80 = abap_true.
        lv_fipex_high = '79'.
      ENDIF.
    ELSE.
      mv_excl80 = abap_false.
    ENDIF.
    "Check if commitment item  is 'NPC'
    READ TABLE it_wfmbdt ASSIGNING <ls_wfmbdt> WITH KEY fipex = 'NPC'.
    IF sy-subrc = 0.
      "Sum amount on NPC for commitment item between 20 and 50 or 80, delete 20 to 50 or 80.
      LOOP AT it_wfmbdt INTO ls_wfmbdt WHERE fipex BETWEEN '20' AND lv_fipex_high.
        ADD ls_wfmbdt-hamount TO <ls_wfmbdt>-hamount.
        ADD ls_wfmbdt-hamount_usd TO <ls_wfmbdt>-hamount_usd.
        DELETE it_wfmbdt.
      ENDLOOP.
      "Do the same for amounts out of selection period
      READ TABLE it_wfmbdt_oth ASSIGNING <ls_wfmbdt_oth> WITH KEY fipex = 'NPC'.
      IF sy-subrc <> 0.
        APPEND INITIAL LINE TO it_wfmbdt_oth ASSIGNING <ls_wfmbdt_oth>.
        <ls_wfmbdt_oth>-fikrs = <ls_wfmbdt>-fikrs.
        <ls_wfmbdt_oth>-fincode = <ls_wfmbdt>-fincode.
        <ls_wfmbdt_oth>-fistl = <ls_wfmbdt>-fistl.
        <ls_wfmbdt_oth>-fipex = 'NPC'.
        <ls_wfmbdt_oth>-waers = <ls_wfmbdt>-waers.
        <ls_wfmbdt_oth>-bukrs = <ls_wfmbdt>-bukrs.
      ENDIF.
      LOOP AT it_wfmbdt_oth INTO ls_wfmbdt_oth WHERE fipex BETWEEN '20' AND lv_fipex_high.
        ADD ls_wfmbdt_oth-hamount TO <ls_wfmbdt_oth>-hamount.
        ADD ls_wfmbdt_oth-hamount_usd TO <ls_wfmbdt_oth>-hamount_usd.
        DELETE it_wfmbdt_oth.
      ENDLOOP.
    ENDIF.

    "Check if commitment item is 'PC'
    READ TABLE it_wfmbdt ASSIGNING <ls_wfmbdt> WITH KEY fipex = 'PC'.
    IF sy-subrc = 0.
      "Sum amount on PC for commitment item between 10 and 19, delete 10 to 19.
      LOOP AT it_wfmbdt INTO ls_wfmbdt WHERE fipex(1) = '1'.
        ADD ls_wfmbdt-hamount TO <ls_wfmbdt>-hamount.
        ADD ls_wfmbdt-hamount_usd TO <ls_wfmbdt>-hamount_usd.
        DELETE it_wfmbdt.
      ENDLOOP.
      "Do the same for amounts out of selection period
      READ TABLE it_wfmbdt_oth ASSIGNING <ls_wfmbdt_oth> WITH KEY fipex = 'PC'.
      IF sy-subrc <> 0.
        APPEND INITIAL LINE TO it_wfmbdt_oth ASSIGNING <ls_wfmbdt_oth>.
        <ls_wfmbdt_oth>-fikrs = <ls_wfmbdt>-fikrs.
        <ls_wfmbdt_oth>-fincode = <ls_wfmbdt>-fincode.

        <ls_wfmbdt_oth>-waers = <ls_wfmbdt>-waers.
        <ls_wfmbdt_oth>-bukrs = <ls_wfmbdt>-bukrs.
      ENDIF.
      LOOP AT it_wfmbdt_oth INTO ls_wfmbdt_oth WHERE fipex(1) = '1'.
        ADD ls_wfmbdt_oth-hamount TO <ls_wfmbdt_oth>-hamount.
        ADD ls_wfmbdt_oth-hamount_usd TO <ls_wfmbdt_oth>-hamount_usd.
        DELETE it_wfmbdt_oth.
      ENDLOOP.
    ENDIF.

    "Set data per commitment item
    LOOP AT it_wfmbdt INTO ls_wfmbdt.
      "Commitment item
      is_result-fipex = ls_wfmbdt-fipex.
      ls_result_usd-fipex = ls_wfmbdt-fipex.
      READ TABLE mt_fmcit INTO DATA(ls_fmcit) WITH KEY fikrs = is_result-fikrs
                                                       fipex = is_result-fipex.
      IF sy-subrc = 0.
        is_result-fipex_txt = ls_fmcit-text1.
        ls_result_usd-fipex_txt = ls_fmcit-text1.
      ENDIF.

      "Support cost %
      CASE ls_wfmbdt-fipex(1).
        WHEN '1' OR 'P'.
          is_result-rate_supcost = ms_project-pour_10.
        WHEN '2'.
          is_result-rate_supcost = ms_project-pour_20.
        WHEN '3'.
          is_result-rate_supcost = ms_project-pour_30.
        WHEN '4'.
          is_result-rate_supcost = ms_project-pour_40.
        WHEN '5'.
          is_result-rate_supcost = ms_project-pour_50.
        WHEN 'T'.
          is_result-rate_supcost = ( ms_project-pour_10 +
                                ms_project-pour_20 +
                                ms_project-pour_30 +
                                ms_project-pour_40 +
                                ms_project-pour_50 ) / 5.
      ENDCASE.
      ls_result_usd-rate_supcost = is_result-rate_supcost.

      "Allocation on the selection period
      is_result-alloc = ls_wfmbdt-hamount.
      is_result-waers = ls_wfmbdt-waers.
      ls_result_usd-alloc = ls_wfmbdt-hamount_usd.
      ls_result_usd-waers = 'USD'.
      "Released budget
      is_result-released = ls_wfmbdt-hamount.
      ls_result_usd-released = ls_wfmbdt-hamount_usd.

      "Total allocation for full life of fund/project
      is_result-alloc_total = is_result-alloc.
      ls_result_usd-alloc_total = ls_result_usd-alloc.
      READ TABLE it_wfmbdt_oth INTO ls_wfmbdt_oth WITH KEY fipex = ls_wfmbdt-fipex.
      IF sy-subrc = 0.

  METHOD sum_hsl_amounts.

    DATA lv_fieldname TYPE fieldname VALUE 'IS_HSL-HSLXX'.
    DATA lv_perio TYPE periode.
    DATA lv_rate TYPE ukurs_curr.
    FIELD-SYMBOLS <field> TYPE hslxx9.

*    DO 17 TIMES.
*      IF sy-index = 1 AND iv_include_cf = abap_false.
*        CONTINUE.
*      ENDIF.
*      ASSIGN COMPONENT sy-index OF STRUCTURE is_hsl TO FIELD-SYMBOL(<lv_amount>).
*      ADD <lv_amount> TO rv_amount.
*    ENDDO.

                  is_hsl-hsl02 +
                  is_hsl-hsl03 +
                  is_hsl-hsl04 +
                  is_hsl-hsl05 +
                  is_hsl-hsl06 +
                  is_hsl-hsl07 +

                  is_hsl-hsl14 +
                  is_hsl-hsl15 +
                  is_hsl-hsl16.
      "Include carry formard ?
      IF iv_include_cf = abap_true.
        ADD is_hsl-hslvt TO rv_amount.
      ENDIF.
    ELSE.
      DO 16 TIMES.
        ADD 1 TO lv_perio.
        lv_fieldname+10 = lv_perio+1(2).
        ASSIGN (lv_fieldname) TO <field>.
        IF <field> IS ASSIGNED.
          IF <field> IS NOT INITIAL.

*            READ TABLE mt_conv_data INTO ls_conv_data WITH KEY waers = mv_waers

*                                                               perio = lv_perio.
            lv_rate = mo_conv_rate->get_rate( iv_waers = mv_waers
                                              iv_gjahr = iv_gjahr
                                              iv_perio = lv_perio ).
            IF lv_rate <> 0.
              rv_amount = rv_amount + ( <field> / lv_rate ).
            ENDIF.
          ENDIF.
        ENDIF.
      ENDDO.
      "Include carry formard ?
      IF iv_include_cf = abap_true.
*        READ TABLE mt_conv_data INTO ls_conv_data WITH KEY waers = mv_waers

        IF lv_rate <> 0.
          rv_amount = rv_amount + ( is_hsl-hslvt / lv_rate ).
        ENDIF.
      ENDIF.
    ENDIF.

    "Inverse sign
    IF iv_inverse = abap_true.
      rv_amount = rv_amount * -1.
    ENDIF.

  ENDMETHOD.

  METHOD set_project_amounts.

    DATA lr_fipex TYPE tty_fipex.
    DATA lv_amount TYPE hamount.
    DATA ls_hsl TYPE ty_hsl.
    DATA lv_rate TYPE ukurs_curr.

    "check first if data found in result table for the project
    READ TABLE mt_result INTO DATA(ls_result) WITH KEY fikrs = iv_fikrs
                                                       fincode = iv_fincode.
    CHECK sy-subrc = 0.

    """"""""""""""""""""""""""""""""""""""""""""""""""
    """""" Set data independant of fund center """""""
    """"""""""""""""""""""""""""""""""""""""""""""""""
    "Approve budget amount
    ms_fipex_all-approuv_usd = ms_project-approv.
    IF ms_fipex_all-approuv_usd IS NOT INITIAL.
      ms_fipex_all-waers_app_usd = ms_project-approv_cuky.
    ENDIF.
    ms_fipex_all-approuv_oth = ms_project-approv_o.
    IF ms_fipex_all-approuv_oth IS NOT INITIAL.
      ms_fipex_all-waers_app_oth = ms_project-approv_o_cuky.
    ENDIF.

    "Allotment current year and cumul
    LOOP AT mt_bpej INTO DATA(ls_bpej) WHERE objnr = ms_project-objnr_prps.
      "Allotment for current year
      IF ls_bpej-gjahr = mv_endye.
        IF ls_bpej-vorga <> 'KBW1'.
          ADD ls_bpej-wljhr TO ms_fipex_all-allot_curr.
          ADD ls_bpej-wljhr TO ms_fipex_all-allot_cumul.
        ENDIF.
      ELSE.
        ADD ls_bpej-wljhr TO ms_fipex_all-allot_cumul.
      ENDIF.
    ENDLOOP.
    "Check if currency conversion to do
    IF mv_convert = abap_true.
      "Convert at sy-datum.
*      READ TABLE mt_conv_data INTO DATA(ls_conv_data) WITH KEY waers = mv_waers
*                                                               gjahr = '0000'
*                                                               perio = '000'.
      lv_rate = mo_conv_rate->get_rate( iv_waers = mv_waers
                                        iv_gjahr = '0000'
                                        iv_perio = 0 ).
      ms_fipex_all-allot_curr = ms_fipex_all-allot_curr / lv_rate.
      ms_fipex_all-allot_cumul = ms_fipex_all-allot_cumul / lv_rate.
      ms_fipex_all-waers_allot = 'USD'.
    ELSE.
      "Associated waers
      ms_fipex_all-waers_allot = ms_project-ledg_waers.
    ENDIF.

    "Revenue, Gain/loss
    lr_fipex = VALUE #( ( sign = 'I' option = 'EQ' low = 'REVENUE' )
                        ( sign = 'I' option = 'EQ' low = 'GAINS' )
                        ( sign = 'I' option = 'EQ' low = 'LOSS' ) ).

    LOOP AT mt_fmit INTO DATA(ls_fmit) WHERE fikrs = iv_fikrs
                                       AND   fincode = iv_fincode
                                       AND   fipex IN lr_fipex.
      IF ls_fmit-wrttp IN mt_wrttp_gr1.
        MOVE-CORRESPONDING ls_fmit TO ls_hsl.
        lv_amount = me->sum_hsl_amounts( is_hsl = ls_hsl iv_gjahr = ls_fmit-ryear iv_convert = mv_convert ).
        CHECK lv_amount IS NOT INITIAL.
        CASE ls_fmit-fipex.
          WHEN 'REVENUE'.
            ADD lv_amount TO ms_fipex_all-revenue.
          WHEN 'GAINS' OR 'LOSS'.
            ADD lv_amount TO ms_fipex_all-gain_loss.
        ENDCASE.
      ENDIF.
    ENDLOOP.

    """""""Compute amounts for commitment item independant
    ms_fipex_all-waers = ls_result-waers.
    "result
    ms_fipex_all-resul = ms_fipex_all-revenue - ms_fipex_all-actual_cumul.
    "Unencombered cash
    ms_fipex_all-unencum = ms_fipex_all-revenue - ms_fipex_all-actual_cumul - ms_fipex_all-oblig_cumul.
    "Available allotment
    ms_fipex_all-allot_avail = ms_fipex_all-allot_curr - ms_fipex_all-oblig_curr - ms_fipex_all-actual_curr - ms_fipex_all-oblig_cf
                               + ms_fipex_all-supp_cost.
    "Rate
    IF ms_fipex_all-allot_curr IS NOT INITIAL.
      "Current year : actual / allotment
      ms_fipex_all-rate_curr_1 = ms_fipex_all-actual_curr * 100 / ms_fipex_all-allot_curr.
      "Current year : ( actual + oblig ) / allotment
      ms_fipex_all-rate_curr_2 = ( ms_fipex_all-actual_curr + ms_fipex_all-oblig_curr ) * 100 / ms_fipex_all-allot_curr.
    ENDIF.
    IF ms_fipex_all-alloc IS NOT INITIAL.
      "Cumul : actual / allocation
      ms_fipex_all-rate_cumul_1 = ms_fipex_all-actual_cumul * 100 / ms_fipex_all-alloc.
      "Cumul : (actual + oblig ) / allocation
      ms_fipex_all-rate_cumul_2 = ( ms_fipex_all-actual_cumul + ms_fipex_all-oblig_cumul ) * 100 / ms_fipex_all-alloc.
    ENDIF.
    IF ms_fipex_all-allot_cumul IS NOT INITIAL.
      "Cumul : actual / allotment
      ms_fipex_all-rate_cumul_3 = ms_fipex_all-actual_cumul * 100 / ms_fipex_all-allot_cumul.
      "Cumul : actual + oblig / allotment
      ms_fipex_all-rate_cumul_4 = ( ms_fipex_all-actual_cumul + ms_fipex_all-oblig_cumul ) * 100 / ms_fipex_all-allot_cumul.
    ENDIF.

    CLEAR: ms_fipex_all-actual_cumul, ms_fipex_all-actual_curr, ms_fipex_all-actual_prev,
           ms_fipex_all-oblig_cumul, ms_fipex_all-oblig_curr, ms_fipex_all-oblig_cf, ms_fipex_all-oblig_futur,
           ms_fipex_all-alloc, ms_fipex_all-alloc_avail,
           ms_fipex_all-supp_cost, ms_fipex_all-released, ms_fipex_all-assigned.

    APPEND ms_fipex_all TO mt_result.

  ENDMETHOD.

  METHOD manage_fmit_data.

    DATA ls_hsl TYPE ty_hsl.
    DATA lv_amount TYPE hamount.
    DATA lv_hslvt TYPE hslvt9.
    DATA lv_amount_usd TYPE hamount.
    DATA lv_hslvt_usd TYPE hslvt9.
    DATA lv_rate TYPE ukurs_curr.

    "Merge period amounts
    MOVE-CORRESPONDING is_fmit TO ls_hsl.
    IF mv_convert = abap_true.
      "Get amounts in USD
      IF is_fmit-wrttp IN mt_wrttp_gr1.   "Convert at period date
        lv_amount_usd = me->sum_hsl_amounts( is_hsl = ls_hsl iv_include_cf = abap_false iv_inverse = abap_true
                                             iv_gjahr = is_fmit-ryear iv_convert = mv_convert ).
*        READ TABLE mt_conv_data INTO ls_conv_data WITH KEY waers = mv_waers
*                                                           gjahr = is_fmit-ryear
*                                                           perio = '000'.
        lv_rate = mo_conv_rate->get_rate( iv_waers = mv_waers
                                          iv_gjahr = is_fmit-ryear
                                          iv_perio = 0 ).
        lv_hslvt_usd = is_fmit-hslvt / lv_rate.
      ELSE.   "Convert at sy-datum
        lv_amount = me->sum_hsl_amounts( is_hsl = ls_hsl iv_include_cf = abap_false iv_inverse = abap_true ).
*        READ TABLE mt_conv_data INTO ls_conv_data WITH KEY waers = mv_waers
*                                                           gjahr = '0000'
*                                                           perio = '000'.
        lv_rate = mo_conv_rate->get_rate( iv_waers = mv_waers
                                          iv_gjahr = '0000'
                                          iv_perio = 0 ).
        lv_amount_usd = lv_amount / lv_rate.
        lv_hslvt_usd = is_fmit-hslvt / lv_rate.
      ENDIF.
    ENDIF.
    "Get Amount in transaction currency
    lv_amount = me->sum_hsl_amounts( is_hsl = ls_hsl iv_include_cf = abap_false iv_inverse = abap_true ).
    lv_hslvt = is_fmit-hslvt.
    CHECK lv_amount IS NOT INITIAL OR lv_hslvt IS NOT INITIAL.
    MULTIPLY lv_hslvt BY -1.
    MULTIPLY lv_hslvt_usd BY -1.
    IF is_fmit-ryear = mv_endye AND is_fmit-btart <> '0300'.
      IF is_fmit-wrttp IN mt_wrttp_gr1.
        ADD lv_amount TO cs_result-actual_curr.
        ADD lv_hslvt TO cs_result-actual_curr.
        ADD lv_amount_usd TO cs_result_usd-actual_curr.
        ADD lv_hslvt_usd TO cs_result_usd-actual_curr.
      ELSE.
        ADD lv_amount TO cs_result-oblig_curr.
        ADD lv_hslvt TO cs_result-oblig_cf.
        ADD lv_amount_usd TO cs_result_usd-oblig_curr.
        ADD lv_hslvt_usd TO cs_result_usd-oblig_cf.
      ENDIF.
    ENDIF.
    IF is_fmit-ryear = mv_prev_year AND is_fmit-btart <> '0300' AND is_fmit-wrttp IN mt_wrttp_gr1.
      ADD lv_amount TO cs_result-actual_prev.
      ADD lv_hslvt TO cs_result-actual_prev.
      ADD lv_amount_usd TO cs_result_usd-actual_prev.
      ADD lv_hslvt_usd TO cs_result_usd-actual_prev.
    ENDIF.
    IF is_fmit-wrttp IN mt_wrttp_gr1.
      ADD lv_amount TO cs_result-actual_cumul.
      ADD lv_hslvt TO cs_result-actual_cumul.
      ADD lv_amount_usd TO cs_result_usd-actual_cumul.
      ADD lv_hslvt_usd TO cs_result_usd-actual_cumul.
    ELSEIF is_fmit-ryear <> mv_endye OR is_fmit-btart <> '0300'.
      ADD lv_amount TO cs_result-oblig_cumul.
      ADD lv_hslvt TO cs_result-oblig_cumul.
      ADD lv_amount_usd TO cs_result_usd-oblig_cumul.
      ADD lv_hslvt_usd TO cs_result_usd-oblig_cumul.
    ENDIF.

  ENDMETHOD.

  METHOD compute_amounts.

    """""""Computed amounts
    "Available allocation
    cs_result-alloc_avail = cs_result-alloc - cs_result-oblig_cumul - cs_result-oblig_futur - cs_result-actual_cumul.
    IF mv_convert = abap_true.
      "Convert amount at sy-datum
*      READ TABLE mt_conv_data INTO DATA(ls_conv_data) WITH KEY waers = mv_waers
*                                                               gjahr = '0000'
*                                                               perio = '000'.
      is_result_usd-alloc_avail = cs_result-alloc_avail / mo_conv_rate->get_rate( iv_waers = mv_waers
                                                                                  iv_gjahr = '0000'
                                                                                  iv_perio = 0 ).
      "Recalculate allocation
      is_result_usd-alloc = is_result_usd-alloc_avail + is_result_usd-oblig_cumul +
                            is_result_usd-oblig_futur + is_result_usd-actual_cumul.
      "Set cnverted amounts to working result
      cs_result = is_result_usd.
    ENDIF.
    "Support cost total
    IF cs_result-fipex = '80'.
      cs_result-supp_cost = cs_result-oblig_cumul + cs_result-actual_curr.
    ENDIF.

    "Assigned
    cs_result-assigned = cs_result-oblig_cumul + cs_result-actual_cumul.
    """""""" Rates per commitment item
    IF cs_result-alloc IS NOT INITIAL.
      "Cumul : actual / allocation
      cs_result-rate_cumul_1 = cs_result-actual_cumul * 100 / cs_result-alloc.
      "Cumul : (actual + oblig ) / allocation
      cs_result-rate_cumul_2 = ( cs_result-actual_cumul + cs_result-oblig_cumul ) * 100 / cs_result-alloc.
    ENDIF.

    "Prepare data for computed amounts independant of commitment item
    ADD cs_result-actual_cumul TO ms_fipex_all-actual_cumul.
    ADD cs_result-oblig_cumul TO ms_fipex_all-oblig_cumul.
    ADD cs_result-actual_curr TO ms_fipex_all-actual_curr.
    ADD cs_result-oblig_curr TO ms_fipex_all-oblig_curr.
    ADD cs_result-alloc TO ms_fipex_all-alloc.
    ADD cs_result-oblig_cf TO ms_fipex_all-oblig_cf.
    ADD cs_result-supp_cost TO ms_fipex_all-supp_cost.
    ADD cs_result-released TO ms_fipex_all-released.
    ADD cs_result-alloc_avail TO ms_fipex_all-alloc_avail.
    ADD cs_result-oblig_futur TO ms_fipex_all-oblig_futur.
    ADD cs_result-assigned TO ms_fipex_all-assigned.
    ADD cs_result-actual_prev TO ms_fipex_all-actual_prev.

  ENDMETHOD.

  METHOD aggregate_data.

    DATA lt_result TYPE TABLE OF ysfm_data_rep_2.
    DATA ls_result_sum TYPE ysfm_data_rep_2.
    DATA lv_fistl TYPE fistl.
    DATA lv_new_fincode TYPE boolean.
    DATA lt_slach TYPE match_result_tab.

    lt_result = mt_result.
    CLEAR mt_result.

    SORT lt_result.

    LOOP AT lt_result INTO DATA(ls_result).

      "Initilization at new fincode
      AT NEW fincode.
        lv_new_fincode = abap_true.
      ENDAT.

      IF lv_new_fincode = abap_true.
        ls_result_sum = ls_result.
        CLEAR ls_result_sum-fistl.
        lv_new_fincode = abap_false.
      ELSE.
        "Fill fund center: take into account when more than one (that should not happen !)
        IF ls_result_sum-fistl IS INITIAL.
          ls_result_sum-fistl = lv_fistl = ls_result-fistl.
        ELSEIF lv_fistl <> ls_result-fistl.
          IF ls_result_sum-fistl+12(4) IS NOT INITIAL.
            FIND ALL OCCURRENCES OF '/' IN ls_result_sum-fistl RESULTS lt_slach.
            LOOP AT lt_slach INTO DATA(ls_slach).
            ENDLOOP.
            IF sy-subrc = 0.
              CLEAR ls_result_sum-fistl+ls_slach-offset.
              ls_result_sum-fistl = |{ ls_result_sum-fistl }/...|.
            ENDIF.
          ELSE.
            ls_result_sum-fistl = |{ ls_result_sum-fistl }/{ ls_result-fistl }|.
          ENDIF.
          lv_fistl = ls_result-fistl.
        ENDIF.
        "Sum amounts depending of commitment items
        ADD ls_result-actual_cumul TO ls_result_sum-actual_cumul.
        ADD ls_result-actual_curr TO ls_result_sum-actual_curr.
        ADD ls_result-oblig_cumul TO ls_result_sum-oblig_cumul.
        ADD ls_result-oblig_curr TO ls_result_sum-oblig_curr.
        ADD ls_result-oblig_cf TO ls_result_sum-oblig_cf.
        ADD ls_result-oblig_futur TO ls_result_sum-oblig_futur.
        ADD ls_result-alloc TO ls_result_sum-alloc.
        ADD ls_result-alloc_avail TO ls_result_sum-alloc_avail.
        ADD ls_result-supp_cost TO ls_result_sum-supp_cost.
        ADD ls_result-released TO ls_result_sum-released.
        ADD ls_result-assigned TO ls_result_sum-assigned.
        ADD ls_result-actual_prev TO ls_result_sum-actual_prev.
      ENDIF.

      AT END OF fincode.
        APPEND ls_result_sum TO mt_result.
      ENDAT.

    ENDLOOP.

  ENDMETHOD.

  METHOD set_main_to_light_table.

    DATA ls_wfmbdt TYPE ty_wfmbdt.
    DATA ls_hsl TYPE ty_hsl.

    ENDIF.

  ENDMETHOD.

  METHOD read_avc.

    CHECK mp_avc = abap_true.

    SELECT SINGLE low FROM tvarvc WHERE name = 'Y_FM_YPS8_FMAVCT_DB_ACCESS'
                                  AND   type = 'P'
           INTO @DATA(lv_value).
    CASE lv_value(1).
      WHEN space.
        SELECT DISTINCT rfikrs AS fikrs,
                        rfund AS fincode,
                        rfundsctr AS fistl,
                        rcmmtitem AS fipex
               FROM fmavct
               FOR ALL ENTRIES IN @mt_fund
               WHERE rldnr = '9H'
               "AND   ryear IN @mr_gjahr
               AND   rfikrs = @mt_fund-fikrs
               AND   rfund = @mt_fund-fincode
               INTO TABLE @mt_fmavct.
      WHEN 'I'.
        SELECT DISTINCT rfikrs AS fikrs,
                        rfund AS fincode,
                        rfundsctr AS fistl,
                        rcmmtitem AS fipex
               FROM fmavct
               FOR ALL ENTRIES IN @mt_fund
               WHERE rfikrs = @mt_fund-fikrs
               AND   rcvrgrp_9 = @space
               AND   rfund = @mt_fund-fincode
               %_HINTS MSSQLNT 'TABLE FMAVCT ABINDEX(2)'
               INTO TABLE @mt_fmavct.
      WHEN 'A'.
        SELECT DISTINCT rfikrs AS fikrs,
                        rfund AS fincode,
                        rfundsctr AS fistl,
                        rcmmtitem AS fipex
               FROM fmavct
               WHERE rldnr = '9H'
               INTO TABLE @mt_fmavct.
        LOOP AT mt_fmavct INTO DATA(ls_fmavct).
          READ TABLE mt_fund TRANSPORTING NO FIELDS WITH KEY fikrs = ls_fmavct-fikrs
                                                             fincode = ls_fmavct-fincode.
          CHECK sy-subrc <> 0.
          DELETE mt_fmavct.
        ENDLOOP.
    ENDCASE.

  ENDMETHOD.
