--- Method Implementation: YCL_FI_ACCOUNT_SUBST_BL=======CM002 ---
  METHOD load_data.

    SELECT DISTINCT a~blart,
                    b~ltext,
                    a~gsber,
                    g~gtext
      FROM ytfi_ba_subst AS a
      LEFT OUTER JOIN t003t AS b ON  b~spras = @sy-langu
                                 AND b~blart = a~blart
      LEFT OUTER JOIN tgsbt AS g ON  g~spras = @sy-langu
                                 AND g~gsber = a~gsber
      WHERE a~bukrs = @mv_bukrs
      INTO CORRESPONDING FIELDS OF TABLE @mt_list.

    LOOP AT mt_list ASSIGNING FIELD-SYMBOL(<ls_list>).
      <ls_list>-hksel = icon_display_more.  "ICON_ENTER_MORE
      SELECT sign,
             opti AS option,
             low,
             high
        FROM ytfi_ba_subst
        WHERE bukrs = @mv_bukrs
        AND   blart = @<ls_list>-blart
        AND   gsber = @<ls_list>-gsber
        AND   sign <> @space
        INTO TABLE @<ls_list>-hktab.
      IF <ls_list>-hktab IS INITIAL.
        <ls_list>-hksel = icon_enter_more.
      ELSE.
        <ls_list>-hksel = icon_display_more.
      ENDIF.
      <ls_list>-hkdis = icon_display.
    ENDLOOP.

  ENDMETHOD.

        lo_column->set_cell_type( if_salv_c_cell_type=>hotspot ).
        lo_column->set_icon( if_salv_c_bool_sap=>true ).
        lo_column->set_long_text( 'G/L Accounts' ).
        lo_column->set_fixed_header_text( 'L' ).

    TRY.

        lo_column->set_icon( if_salv_c_bool_sap=>true ).

--- Method Implementation: YCL_FI_ACCOUNT_SUBST_BL=======CM00B ---
  METHOD update_data.

    DATA lt_data TYPE TABLE OF ytfi_ba_subst.
    DATA ls_data TYPE ytfi_ba_subst.
    DATA lv_numb TYPE ytfi_ba_subst-numb.

    "First delete existing entries
    DELETE FROM ytfi_ba_subst WHERE bukrs = mv_bukrs
                              AND   blart = iv_blart
                              AND   gsber = iv_gsber.
    ls_data-bukrs = mv_bukrs.
    ls_data-blart = iv_blart.
    ls_data-gsber = iv_gsber.

    LOOP AT it_hkont INTO DATA(ls_hkont).
      ADD 1 TO lv_numb.
      ls_data-numb = lv_numb.
      ls_data-sign = ls_hkont-sign.
      ls_data-opti = ls_hkont-option.
      ls_data-low = ls_hkont-low.
      ls_data-high = ls_hkont-high.
      APPEND ls_data TO lt_data.
    ENDLOOP.
    IF sy-subrc <> 0.
      APPEND ls_data TO lt_data.
    ENDIF.

    INSERT ytfi_ba_subst FROM TABLE lt_data.
    IF sy-subrc <> 0.
      MESSAGE TEXT-e02 TYPE 'E'.
      ROLLBACK WORK.
    ELSE.
      COMMIT WORK.
    ENDIF.

  ENDMETHOD.

--- Method Implementation: YCL_FI_ACCOUNT_SUBST_BL=======CM00G ---
  METHOD check_accounts.

    DATA lt_acct_all TYPE TABLE OF ty_acct.
    DATA ls_check TYPE ty_check.

    CLEAR mt_check.

    LOOP AT mt_list INTO DATA(ls_list).
      IF ls_list-hktab IS INITIAL.
        CLEAR ls_check.
        MOVE-CORRESPONDING ls_list TO ls_check.
        ls_check-messg = TEXT-m02.
        APPEND ls_check TO mt_check.
      ELSE.
        me->read_account_list( ls_list ).
        APPEND LINES OF mt_acct TO lt_acct_all.
      ENDIF.
    ENDLOOP.

    SORT lt_acct_all BY blart hkont.

    LOOP AT lt_acct_all INTO DATA(ls_acct_all) GROUP BY ( blart = ls_acct_all-blart
                                                          saknr = ls_acct_all-hkont
                                                          size = GROUP SIZE )
                        ASSIGNING FIELD-SYMBOL(<ls_acct_group>).
      IF <ls_acct_group>-size > 1.
        LOOP AT GROUP <ls_acct_group> INTO DATA(ls_acct_group).
          CLEAR ls_check.
          MOVE-CORRESPONDING ls_acct_group TO ls_check.
          ls_check-messg = TEXT-m01.
          APPEND ls_check TO mt_check.
        ENDLOOP.
      ENDIF.
    ENDLOOP.

    mo_alv_check->execute( EXPORTING iv_title = 'Inconsistency in G/L Accounts assignment'
                           CHANGING ct_table = mt_check ).

  ENDMETHOD.


--- Method Implementation: YCL_FI_ACCOUNT_SUBST_BL=======CM00H ---
  METHOD read_account_list.

    CLEAR mt_acct.

    SELECT a~bukrs, @is_list-blart, @is_list-gsber, a~saknr, t~txt50
       FROM skb1 AS a
       INNER JOIN t001 AS b ON b~bukrs = a~bukrs
       LEFT OUTER JOIN skat AS t ON  t~spras = @sy-langu
                                 AND t~ktopl = b~ktopl
                                 AND t~saknr = a~saknr
       WHERE a~bukrs = @mv_bukrs
       AND   a~saknr IN @is_list-hktab
       INTO  TABLE @mt_acct.

  ENDMETHOD.
